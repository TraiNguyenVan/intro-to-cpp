switch (m) {
case Month::feb:
ndays = 28+is_leapyear(y);
break;
case Month::apr: case Month::jun: case Month::sep: case Month::nov:

Tính mơ hồ (ambiguity)
Phép gán một giá trị của kiểu V vào một đối tượng thuộc kiểu lớp X là hợp lệ nếu có một toán tử gán X::operator=(Z) sao cho V là Z hoặc có một phép chuyển đổi duy nhất từ V thành Z.
Nếu một giá trị của kiểu chuyển đến có thể được khởi tạo bằng nhiều cách -> không hợp lệ.
Do đó, không định nghĩa phép chuyển kiểu bằng cả hai cách: dùng hàm tạo constructor và dùng hàm toán tử chuyển đổi.

Tính mơ hồ (ambiguity) – Ví dụ 01
class X { /* ... */ X(int); X(const char∗); };
class Y { /* ... */ Y(int); };
class Z { /* ... */ Z(X); };
X f(X);
Y f(Y);
Z g(Z);
void k1()
{
f(1); // error: ambiguous f(X(1)) or f(Y(1))?
f(X{1}); // OK
f(Y{1}); // OK
g("Mack"); // error: two user-defined conversions needed; g(Z{X{"Mack"}}) not tried
g(X{"Doc"}); // OK: g(Z{X{"Doc"}})
g(Z{"Suzy"}); // OK: g(Z{X{"Suzy"}})
}

[]
Hàm toán tử operator[]() phải là hàm thành viên non-static.
struct Assoc {
vector<pair<string,int>> vec; // vector of {name,value} pairs
const int& operator[] (const string&) const;
int& operator[](const string&);
};
int& Assoc::operator[](const string& s)
// search for s; return a reference to its value if found;
// otherwise, make a new pair {s,0} and return a reference to its value
{
for (auto x : vec)
if (s == x.first) return x.second;
vec.push_back({s,0}); // initial value: 0
return vec.back().second; // return last element
}
int main() // count the occurrences of each word on input
{
Assoc values;
string buf;
while (cin>>buf) ++values[buf];
for (auto x : values.vec)
cout << '{' << x.first << ',' << x.second << "}\n";
}

++ và -- 
Định nghĩa lại các phép ++ và -- trong cả hai trường hợp dùng: hậu tố và tiền tố.
Phép cộng-tiền tố có thể trả về một tham chiếu đến đối tượng của nó.

Hàm friend
Một khai báo hàm thành viên thông thường có 3 đặc điểm sau:
1. Hàm có thể truy xuất đến phần riêng (private) của khai báo lớp
2. Hàm nằm trong phạm vi (scope) của lớp
3. Hàm phải được gọi theo đối tượng của lớp (có con trỏ this)
• Nếu khai báo hàm là static, thì hàm chỉ có (1) và (2)
• Nếu khai báo một hàm không phải thành viên là friend, hàm này chỉ có (1)
i.e., hàm friend được cấp quyền truy xuất đến phần cài đặt của lớp tương đương như vai trò hàm thành viên của lớp nhưng lại độc lập với lớp này.

Ví dụ:
constexpr rc_max {4}; // row and column size
class Matrix;
class Vector {
float v[rc_max];
// ...
friend Vector operator∗(const Matrix&, const Vector&);
};
class Matrix {
Vector v[rc_max];
// ...
friend Vector operator∗(const Matrix&, const Vector&);
};
Vector operator∗(const Matrix& m, const Vector& v)
{
Vector r;
for (int i = 0; i!=rc_max; i++) { // r[i] = m[i] * v;
r.v[i] = 0;
for (int j = 0; j!=rc_max; j++)
r.v[i] += m.v[i].v[j] ∗ v.v[j];
}
return r;
}

Hàm friend – các đặc tính
• Khai báo friend có thể đặt ở phần private hoặc phần public 
• Hàm friend được khai báo tường minh trong lớp “bạn” của hàm
• Hàm friend của một lớp có thể là friend của một lớp khác.
Ví dụ: 
class List_iterator {
// ...
int∗ next();
};
class List {
friend int∗ List_iterator::next();
// ...
};

Khai báo friend cho một lớp
• Cấp quyền truy xuất đến tất cả các hàm của lớp đó.
class List {
friend class List_iterator;
// ...
};
• Đối số mẫu thành friend 
template<typename T>
class X {
friend T;
friend class T; // redundant “class”
// ...
};

Ví dụ
class C1 { }; // will become friend of N::C
void f1(); // will become friend of N::C
namespace N {
class C2 { }; // will become friend of C
void f2() { } // will become friend of C
class C {
int x;
public:
friend class C1; // OK (previously defined)
friend void f1();
friend class C3; // OK (defined in enclosing namespace)
friend void f3();
friend class C4; // First declared in N and assumed to be in N
friend void f4();
};
class C3 {}; // friend of C
void f3() { C x; x.x = 1; } // OK: friend of C
} // namespace N
class C4 { }; // not friend of N::C
void f4() { N::C x; x.x = 1; } // error: x is private and f4() is not a friend of N::C

Ví dụ: minh họa các cách viết hàm toán tử với khai báo friend và khai báo thành viên.
class X {
// ...
X(int);
int m1(); // member
int m2() const;
friend int f1(X&); // friend, not member
friend int f2(const X&);
friend int f3(X);
};

Các hàm friend và hàm thành viên
• Hàm thành viên chỉ được gọi khi áp dụng cho các đối tượng thuộc lớp chứa hàm. Không áp dụng cách chuyển kiểu do người dùng định nghĩa cho toán hạng chính của phép tham chiếu . Và -> .
Ví dụ:
void g()
{
99.m1(); // error: X(99).m1() not tried
99.m2(); // error: X(99).m2() not tried
}
• Chuyển kiểu không tường minh không được áp dụng cho các đối số tham chiếu non-const.
-> Toán tử thay đổi trạng thái của đối tượng một lớp là thành viên hoặc là hàm nhận đối số tham chiếu non-const (hoặc đối số là con trỏ tới non-const) 
void h()
{
f1(99); // error: f1(X(99)) not tried: non-const X& argument
f2(99); // OK: f2(X(99)); const X& argument
f3(99); // OK: f3(X(99)); X argument
}

Lớp Dẫn xuất và lớp Cơ sở
Viết chương trình quản lý thông tin nhân viên một công ty
• Cấu trúc dữ liệu nhân viên
struct Employee {
string first_name, family_name;
char middle_initial;
Date hiring_date;

// ...
};
• Định nghĩa người quản lý:
struct Manager {
Employee emp; // manager’s employee record
list<Employee∗> group; // people managed
short level;
// ...
};

Lớp Dẫn xuất và lớp Cơ sở
Cần một khai báo rõ ràng rằng người quản lý Manager cũng là một nhân viên Employee 
struct Manager : public Employee {
list<Employee∗> group;
short level;
// ...
};
• Manager được dẫn xuất (derived) từ Employee. Hay phát biểu ngược lại, Employee là lớp cơ sở (base class) cho Manager.
• Thành viên của lớp dẫn xuất: Lớp Manager có 
• Các thành viên của lớp Employee (first_name, department, v.v.) và
• Các thành viên riêng của nó (group, level, v.v.) 

Sự Kế thừa (Inheritance)
• Lớp dẫn xuất kế thừa các thuộc tính từ lớp cơ sở của nó.
• Lớp cơ sở : siêu lớp (superclass)
• Lớp dẫn xuất : lớp con (subclass)
• Lớp dẫn xuất thường lớn hơn lớp cơ sở của nó theo nghĩa nó chứa nhiều dữ liệu hơn và cung cấp nhiều chức năng hơn
Employee -> Manager
Employee: first_name, family_name, ...
Manager: first_name, family_name, ..., group, level, ...

Sử dụng
void f(Manager m1, Employee e1)
{
list<Employee∗> elist {&m1,&e1};
// ...
}
Vì một Manager cũng là một Employee , 
-> một con trỏ Manager* có thể được sử dụng như con trỏ Employee*
-> Manager& có thể được sử dụng như Employee&
Điều ngược lại, một Employee không nhất thiết là một Manager
-> Employee* không thể được sử dụng như Manager* .
Tổng quát:
• Nếu một lớp dẫn xuất Derived có lớp cơ sở công khai là Base, thì Derived* có thể được gán cho một biến kiểu Base* mà không cần dùng tới phép chuyển kiểu tường minh.
• Phép chuyển Base* thành Derived* phải là tường minh

Ví dụ
void g(Manager mm, Employee ee)
{
Employee∗ pe = &mm; // OK: every Manager is an Employee
Manager∗ pm = &ee; // error: not every Employee is a Manager
pm−>level = 2; // disaster: ee doesn’t have a level
pm = static_cast<Manager∗>(pe); // brute force: works because pe points
// to the Manager mm
pm−>level = 2; // fine: pm points to the Manager mm that has a level
}
Chú ý: Một đối tượng của lớp dẫn xuất có thể được xem là đối tượng của lớp cơ sở thông qua con trỏ và tham chiếu.
Ngược lại là không đúng!

Các hàm thành viên trong kế thừa
class Employee {

// ...
private:

// ...
};
class Manager : public Employee {
public:

// ...
};

Các hàm thành viên trong kế thừa
• Một thành viên của lớp dẫn xuất có thể sử dụng các thành viên public và protected của lớp cơ sở nếu các thành viên này được khai báo trong lớp dẫn xuất.
void Manager::print() const
{
cout << "name is " << full_name() << '\n';
// ...
}
• Lớp dẫn xuất không truy xuất được các thành viên private của lớp cơ sở.
void Manager::print() const
{
cout << " name is " << family_name << '\n'; // error!
// ...
}
-> Có thể gọi các thành viên public của lớp cơ sở
void Manager::print() const
{
Employee::print(); // print Employee information
cout << level; // print Manager-specific information
// ...
}

Hàm khởi tạo và Hàm hủy trong kế thừa
• Hàm khởi tạo và hàm hủy của lớp cơ sở không được kế thừa.
• Các đối tượng được tạo từ dưới lên (bottom up): đối tượng cơ sở -> thành viên -> đối tượng lớp dẫn xuất.
Các đối tượng được hủy từ trên xuống (top down): đối tượng lớp dẫn xuất -> thành viên -> đối tượng cơ sở;
• Mỗi lớp có thể khởi tạo các thành viên của mình (nhưng không khởi tạo trực tiếp các thành viên hoặc lớp cơ sở của lớp cơ sở của nó);
• Các destructors trong phân cấp kế thừa cần được khai báo virtual;
• Copy constructors của các lớp trong phân cấp kế thừa nên được dùng cẩn thận để tránh gây cắt lát;
• Lời gọi hàm ảo với dynamic_cast hoặc typeid() trong constructor/destructor phản ánh một giai đoạn của việc tạo và hủy đối tượng

Phân cấp lớp
• Một lớp dẫn xuất có thể là một lớp cơ sở.
class Employee { /* ... */ };
class Manager : public Employee { /* ... */ };
class Director : public Manager { /* ... */ };
• Các lớp kế thừa nhau tạo thành kiến trúc phân cấp lớp
class Temporary { /* ... */ };
class Assistant : public Employee { /* ... */ };
class Temp : public Temporary, public Assistant { /* ... */ };
class Consultant : public Temporary, public Manager { /* ... */ };
(Sơ đồ kế thừa: Employee <- Assistant, Manager. Temporary, Assistant <- Temp. Temporary, Manager <- Consultant. Manager <- Director)

Hàm ảo
Cho phép khai báo hàm trong lớp cơ sở và sau đó có thể định nghĩa lại hàm (ghi đè – override) trong lớp dẫn xuất.
• Trình biên dịch và trình liên kết sẽ đảm bảo sự tương ứng chính xác giữa đối tượng và các hàm được áp dụng cho đối tượng.

Hàm ảo – Ví dụ
virtual void print() const :
• print() đóng vai trò là giao diện cho hàm print() được định nghĩa trong lớp Employee và các hàm print() được định nghĩa trong các lớp dẫn xuất từ Employee.
class Employee {
public:
Employee(const string& name, int dept);
virtual void print() const;
// ...
private:
string first_name, family_name;
short department;
// ...
};

Hàm ảo – Quy tắc Khai báo
• Kiểu của đối số trong khai báo hàm ảo ở lớp dẫn xuất phải tương đồng với khai báo hàm trong lớp cơ sở.
• Hàm ảo phải được định nghĩa ở lớp cơ sở nơi nó được khai báo đầu tiên.
Ví dụ: 
void Employee::print() const
{
cout << family_name << '\t' << department << '\n';
// ...
}

Hàm ảo – Quy tắc Khai báo
• Hàm ảo có thể được sử dụng không phụ thuộc vào lớp có các lớp dẫn xuất hay không.
• Nếu lớp dẫn xuất không cần dùng hàm ảo thì có thể bỏ qua không cần khai báo hàm này. Chỉ cung cấp hàm nếu cần thiết.
class Manager : public Employee {

// ...
};
void Manager::print() const

// ...
}

Hàm ghi đè (override functions)
Một hàm trong lớp dẫn xuất có cùng tên và cùng tập các kiểu đối số (chữ ký của hàm) với hàm ảo trong lớp cơ sở được gọi là ghi đè lên phiên bản ở lớp cơ sở của hàm ảo.
• Hàm ghi ghi đè được phép trả về một kiểu là biến thể của kiểu trả về nguyên thủy.
• Lời gọi hàm ảo có thể thực hiện tường minh tới hàm ảo với tham chiếu từ lớp sở hữu (ví dụ Employee::print() ), hoặc gọi tới hàm ghi đè tương ứng với đối tượng đang hoạt động.

Hàm ghi đè – Ví dụ
void print_list(const list<Employee∗>& s)
{
for (auto x : s)
x−>print();
}
int main()
{
Employee e {"Brown",1234};
Manager m {"Smith",1234,2};
print_list({&e,&m});
}
Kết quả:
Smith 1234
level 2
Brown 1234

Đa hình - Polymorphism
Là việc đối tượng (của một lớp) có các hành vi khác nhau trong các tình huống khác nhau.
• Xác định hành vi đúng của một lớp độc lập với kiểu lớp nào được sử dụng.
• Khai báo hàm ảo được gọi là kiểu đa hình, hay còn gọi là kiểu đa hình khi thực thi chương trình (run-time polymorphic type).

Cách gọi hàm trong cơ chế đa hình
Trình biên dịch chuyển đổi tên của một hàm ảo thành một chỉ mục (index) trong một bảng các con trỏ trỏ tới các hàm. Bảng này gọi là bảng hàm ảo (vtbl).
Mỗi lớp có các hàm ảo đều có vtbl riêng để xác định các hàm ảo của nó.
Employee: first_name, second_name, vtbl -> Employee::print()
Manager: first_name, second_name, group, level, vtbl -> Manager::print()

Các loại đa hình
• Đa hình khi biên dịch (compile-time polymorphism)
• Cơ chế chồng hàm (function overloading): các hàm cùng tên, khác tham số.
• Trình biên dịch xác định lời gọi hàm cụ thể để liên kết với đối tượng sau khi suy ra đối tượng tại thời điểm chạy.
• Đa hình khi thực thi (runtime polymorphism)
• Cơ chế ghi đè hàm (function overriding): ở lớp dẫn xuất có định nghĩa lại một hoặc nhiều hàm ở lớp cơ sở.
• Chỉ xác định được lời gọi hàm tại thời điểm thực thi.
• Còn gọi là kết nối muộn (late binding) hay đa hình động

Kiểm soát cơ chế ghi đè
Sử dụng các chỉ định sau:
• virtual: cho biết hàm có thể được ghi đè.
• =0: Hàm phải được khai báo virtual và phải được ghi đè.
• override: Hàm phải ghi đè một hàm ảo trong lớp cơ sở.
• final: Hàm không được phép ghi đè.
Nếu không dùng các chỉ định trên, một hàm thành viên non-static sẽ là hàm ảo nếu và chỉ nếu nó ghi đè một hàm được chỉ định virtual trong lớp cơ sở.

Chỉ định override 
Chỉ định override không phải là một phần của kiểu hàm và không được lặp lại trong định nghĩa hàm nằm bên ngoài lớp.
class Derived : public Base {
void f() override; // OK if Base has a virtual f()
void g() override; // OK if Base has a virtual g()
};
void Derived::f() override // error: override out of class
{
// ...
}
void g() // OK
{
// ...
}

Chỉ định final 
Để dừng ý định cho phép ghi đè lên hàm virtual.
struct Node { // interface class
virtual Type type() = 0;
// ...
};
class If_statement : public Node {
public:
Type type() override final; // prevent further overriding
// ...
};
class Modified_if_statement : public If_statement {
public:
Type type() override; // error: If_statement::type() is final
// ...
};

Chỉ định final 
Để dừng ý định cho phép ghi đè lên hàm virtual. Áp dụng cho toàn bộ các hàm trong lớp.
class For_statement final : public Node {
public:
Type type() override;
// ...
};
class Modified_for_statement : public For_statement { // error: For_statement is final
Type type() override;
// ...
};

Chỉ định final 
Chỉ định final không phải là một phần của kiểu hàm và không được lặp lại trong định nghĩa hàm nằm bên ngoài lớp.
class Derived : public Base {
void f() final; // OK if Base has a virtual f()
void g() final; // OK if Base has a virtual g()
// ...
};
void Derived::f() final // error: final out of class
{
// ...
}
void g() final // OK
{
// ...
}

Bổ sung các thành viên lớp cơ sở bằng using 
Các hàm không ghi chồng vượt phạm vi của lớp.
Ví dụ:
struct Base {
void f(int);
};
struct Derived : Base {
void f(double);
};
void use(Derived d)
{
d.f(1); // call Derived::f(double)
Base& br = d
br.f(1); // call Base::f(int)
}

Bổ sung các thành viên lớp cơ sở bằng using 
Có thể bổ sung hàm vào lớp bằng khai báo using .
-> Bổ sung vào không gian tên (namespace) của lớp.
Ví dụ:
struct Base {
void f(int);
};
struct D2 : Base {
using Base::f; // bring all fs from Base into D2
void f(double);
};
void use2(D2 d)
{
d.f(1); // call D2::f(int), that is, Base::f(int)
Base& br = d
br.f(1); // call Base::f(int)
}

Bổ sung các thành viên lớp cơ sở bằng using 
Có thể bổ sung các hàm từ nhiều lớp cơ sở.
Ví dụ:
struct B1 {
void f(int);
};
struct B2 {
void f(double);
};
struct D : B1, B2 {
using B1::f;
using B2::f;
void f(char);
};
void use(D d)
{
d.f(1); // call D::f(int), that is, B1::f(int)
d.f('a'); // call D::f(char)
d.f(1.0); // call D::f(double), that is, B2::f(double)
}

Kế thừa các hàm tạo
Ví dụ: tạo một lớp vector mới tương tự std::vector nhưng có thêm phép kiểm tra miền giá trị kích thước vector.
template<class T>
struct Vector : std::vector<T> {
T& operator[](size_type i) { check(i); return this−>elem(i); }
const T& operator[](size_type i) const { check(i); return this−>elem(i); }
void check(size_type i) { if (this−>size()<i) throw range_error{"Vector::check() failed"}; }
};
Khai báo này đã không kế thừa các hàm tạo từ std::vector, nên gây lỗi khởi tạo vector.
Vector<int> v { 1, 2, 3, 5, 8 }; // error: no initializer-list constructor

Kế thừa các hàm tạo
template<class T>
struct Vector : std::vector<T> {
using vector<T>::vector; // inherit constructors
T& operator[](size_type i) { check(i); return this−>elem(i); }
const T& operator[](size_type i) const { check(i); return this−>elem(i); }
void check(size_type i) { if (this−>size()<i) throw Bad_index(i); }
};
Vector<int> v { 1, 2, 3, 5, 8 }; // OK: use initializer-list constructor from std::vector

Kế thừa các hàm tạo - Cẩn thận khi khởi tạo biến thành viên trong lớp kế thừa
struct B1 {
B1(int) { }
};
struct D1 : B1 {
using B1::B1; // implicitly declares D1(int)
string s; // string has a default constructor
int x; // we “forgot” to provide for initialization of x
};
void test()
{
D1 d {6}; // oops: d.x is not initialized
D1 e; // error: D1 has no default constructor
}

Kế thừa các hàm tạo - Cẩn thận khi khởi tạo biến thành viên trong lớp kế thừa
Cách khắc phục
struct B1 {
B1(int) { }
};
struct D1 : B1 {
D1(int i) : B1(i) { }
string s; // string has a default constructor
int x; // we “forgot” to provide for initialization of x
};
struct B1 {
B1(int) { }
};
struct D1 : B1 {
using B1::B1; // implicitly declares D1(int)
int x {0}; // note: x is initialized
};
void test()
{
D1 d {6}; // d.x is zero
}

Các Lớp Trừu tượng
Abstract Classes

Các khái niệm và Định nghĩa
• Một hàm ảo được định nghĩa “thuần ảo” bằng chỉ định =0.
Một lớp có một hoặc nhiều hàm ảo thuần được gọi là lớp trừu tượng.
• Lớp trừu tượng là giao diện cho các đối tượng được truy cập thông qua con trỏ và tham chiếu (nhằm bảo toàn tính chất đa hình).
-> Lớp trừu tượng có một hàm hủy ảo.
• Không có đối tượng nào được tạo ra từ lớp trừu tượng
-> Lớp trừu tượng không có hàm tạo

Ví dụ AC_01
class Shape { // abstract class
public:
virtual void rotate(int) = 0; // pure virtual function
virtual void draw() const = 0; // pure virtual function
virtual bool is_closed() const = 0; // pure virtual function
// ...
virtual ˜Shape(); // virtual
};

Ví dụ AC_02
• Lớp trừu tượng chỉ có thể được sử dụng như là một giao diện cho các lớp khác.
class Point { /* ... */ };
class Circle : public Shape {
public:
void rotate(int) override { }
void draw() const override;
bool is_closed() const override { return true; }
Circle(Point p, int r);
private:
Point center;
int radius;
};

Lớp dẫn xuất trừu tượng 
• Một hàm thuần ảo không được định nghĩa lại ở lớp kế thừa (là lớp dẫn xuất) vẫn giữ nguyên tính chất hàm thuần ảo
-> Lớp dẫn xuất cũng là lớp trừu tượng.
Ví dụ: 
class Polygon : public Shape { // abstract class
public:
bool is_closed() const override { return true; }
// ... draw and rotate not overridden ...
};
Polygon b {p1,p2,p3,p4}; // error: declaration of object of abstract class Polygon

Khai báo khắc phục lỗi cho Polygon
class Irregular_polygon : public Polygon {
list<Point> lp;
public:
Irregular_polygon(initializer_list<Point>);

Kiểm soát Truy cập vào thành viên của lớp

Phân loại thành viên theo khả năng truy cập
Các thành viên có thể được khai báo private, protected, hoặc public.
• private: chỉ được truy cập và sử dụng bởi các hàm thành viên và các hàm bạn của lớp chứa.
• protected: có thể được truy cập và sử dụng bởi các hàm thành viên và các hàm bạn của lớp chứa, và các hàm thành viên và bạn của các lớp dẫn xuất từ lớp này.
• public: được truy cập và sử dụng tự do.
general users -> public:
derived class’s member functions and friends -> protected:
own member functions and friends -> private:

Truy cập đến các lớp cơ sở
Các lớp cơ sở được khai báo private, protected, hoặc public.
class X : public B { /* ... */ };
class Y : protected B { /* ... */ };
class Z : private B { /* ... */ };

Truy cập đến các lớp cơ sở
Giả sử lớp D dẫn xuất (kế thừa) B. 
Khai báo lớp cơ sở B là:
• private: các thành viên public và protected của B chỉ được sử dụng bởi các hàm thành viên và bạn của D. Chỉ các thành viên và bạn của D mới đổi được D* thành B*.
• protected: các thành viên public và protected của B chỉ được sử dụng bởi các hàm thành viên và bạn của D, và các hàm thành viên và bạn của các lớp kế thừa từ D. Chỉ các thành viên và bạn của D và thành viên và bạn của các lớp kế thừa từ D mới đổi D* thành B*.
• public: các thành viên public của B được sử dụng bởi hàm bất kỳ. Thành viên protected có thể được sử dụng bởi các thành viên và bạn của D và thành viên và bạn của các lớp kế thừa từ D. Bất kỳ hàm nào cũng có thể đổi D* thành B*.

Đa kế thừa
struct B {
int m;
static int sm;
// ...
};
class D1 : public virtual B { /* ... */ } ;
class D2 : public virtual B { /* ... */ } ;
class D12 : public D1, private D2 { /* ... */ };
D12∗ pd = new D12;
B∗ pb = pd; // OK: accessible through D1
int i1 = pd−>m; // OK: accessible through D1

Khai báo using và kiểm soát truy cập
class B {
private:
int a;
protected:
int b;
public:
int c;
};
class D : public B {
public:
using B::a; // error: B::a is private
using B::b; // make B::b publicly available through D
};

Khai báo using và kiểm soát truy cập
class BB : private B { // give access to B::b and B::c, but not B::a
public:
using B::b;
using B::c;
};

Con trỏ tới các thành viên của lớp
Các toán tử truy xuất ->* và .* 

Khai báo con trỏ tới thành viên lớp
• Dùng các toán tử ->* và .*
• Con trỏ tới thành viên không thể gán tới void* hay các kiểu con trỏ thông thường.
• Con trỏ rỗng có thể được gán cho con trỏ tới thành viên. Ý nghĩa: “không thành viên”

Con trỏ tới các thành viên hàm
Ví dụ:
1. Khai báo lớp giao diện tiêu chuẩn cho các hàm chức năng
2. Xử lý cụ thể của các chức năng sẽ được định nghĩa bởi đối tượng của lớp dẫn xuất, kế thừa từ Std_interface.
class Std_interface {
public:
virtual void start() = 0;
virtual void suspend() = 0;
virtual void resume() = 0;
virtual void quit() = 0;
virtual void full_size() = 0;
virtual void small() = 0;
virtual ˜Std_interface() {}
};

Con trỏ tới các thành viên hàm
• Dùng toán tử “địa chỉ của” - & - trỏ tới tên thành viên cụ thể của lớp.
Vd: &Std_interface::suspend.
• Khai báo biến kiểu con trỏ trỏ tới thành viên của lớp X: 
 X::*
using Pstd_mem = void (Std_interface::∗)(); // pointer-to-member type
void f(Std_interface∗ p)
{
Pstd_mem s = &Std_interface::suspend; // pointer to suspend()
p−>suspend(); // direct call
p−>∗s(); // call through pointer to member
}

Con trỏ tới các thành viên hàm
• Toán tử ->* và .* cho phép một con trỏ tới thành viên có thể được dùng kết hợp với một một đối tượng.
Vd: 
 p->*m liên kết m với đối tượng được trỏ bởi p. 
 obj.*m liên kết m với đối tượng obj.
• Kết quả trả về của con trỏ tới thành viên hàm có kiểu tương ứng với kiểu trả về của hàm đã được định nghĩa.
• Chú ý: Không thể lưu lại kết quả của ->* hay .* để dùng sau đó!!

Con trỏ tới các thành viên hàm
Lời gọi p->*s() :
p -> vtbl -> X::start, X::suspend (s chỉ mục tới X::suspend)

Con trỏ tới các thành viên hàm – Các lưu ý (1/2)
• Hàm được gọi qua con trỏ hàm có thể là hàm virtual.
Ví dụ: Nếu gọi tới hàm suspend() bằng một con trỏ hàm, sẽ được hàm suspend() đúng theo đối tượng mà con trỏ tới hàm đang áp dụng.
Ví dụ: Trong ví dụ sau dùng các con trỏ tới hàm thành viên để gọi các hàm được biểu diễn bằng các chuỗi “tên hàm” tương ứng.
map<string,Std_interface∗> variable;
map<string,Pstd_mem> operation;
void call_member(string var, string oper)
{
(variable[var]−>∗operation[oper])(); // var.oper()
}

Con trỏ tới các thành viên hàm – Các lưu ý (2/2)
• Vì hàm thành viên khai báo static không có liên quan đến một đối tượng cụ thể nào, nên con trỏ tới một thành viên static chỉ được xem là một con trỏ thông thường.
Ví dụ: 
class Task {
// ...
static void schedule();
};
void (∗p)() = &Task::schedule; // OK
void (Task::∗ pm)() = &Task::schedule; // error: ordinary pointer assigned
// to pointer to member

Con trỏ tới các thành viên dữ liệu
Cho khai báo một lớp cấu trúc C như sau:

using Pmfi = void (C::∗)(int); // pointer to member function of C taking an int
using Pm = const char∗ C::∗; // pointer to char* data member of C

Con trỏ tới các thành viên dữ liệu
Hãy phân tích đoạn chương trình:

{
C∗ p = &z2;
Pmfi pf = &C::print;

(z1.∗pf)(2);
z1.∗pm = "nv1 ";
p−>∗pm = "nv2 ";

(p−>∗pf)(4);
pf = &C::f1; // error: return type mismatch
pf = &C::f2; // error: argument type mismatch
pm = &C::i; // error: type mismatch
pm = pf; // error: type mismatch
}

• Và thường sẽ bổ sung thêm các thành viên riêng của nó.

• Trong kiến trúc lớp và lớp thành viên
• Constructor tạo đối tượng lớp theo trình tự từ dưới lên:
1. Constructor gọi các contructors của lớp cơ sở
2. Constructor tiếp tục gọi các constructor thành viên

• Destructor hủy đối tượng lớp từ trên xuống:

2. Destructor gọi các destructor thành viên
3. Destructor gọi các destructor của lớp cơ sở
• Đảm bảo nguyên tắc: chỉ được sử dụng khi đã khởi tạo đầy đủ và trước khi các tài nguyên bị hủy.

• Destructor được gọi ngầm định khi thoát khỏi một phạm vi truy xuất đối tượng, hoặc thực hiện xóa đối tượng bằng lệnh delete.
• Nếu constructor tạo đối tượng với yêu cầu tạo mới tài nguyên (new), ta sẽ cần lời gọi tường minh tới destructor
void C::push_back(const X& a)
{
// ...
new(p) X{a}; // copy construct an X with the value a in address p
// ...
}
void C::pop_back()
{
// ...
p−>˜X(); // destroy the X in address p
}

Khởi tạo Đối tượng bằng Constructors
• Phải khởi tạo đối tượng bằng lời gọi đến hàm constructor phù hợp
struct X {
X(int);
};
X x0; // error: no initializer
X x1 {}; // error: empty initializer
X x2 {2}; // OK
X x3 {"two"}; // error: wrong initializer type
X x4 {1,2}; // error: wrong number of initializers
X x5 {x4}; // OK: a copy constructor is implicitly defined

Khởi tạo Đối tượng bằng Constructors với {}
• Constructor mặc định sẽ không có tác dụng nếu đã khai báo constructor với các đối số
• Copy constructor được duy trì để đảm bảo các đối tượng có thể được sao chép
• Nên sử dụng {} để tường minh việc khởi tạo.
struct Y : X {
X m {0}; // provide default initializer for member m
Y(int a) :X{a}, m{a} { }; // initialize base and member
Y() : X{0} { }; // initialize base and member
};
X g {1}; // initialize global variable
void f(int a)
{
X def {}; // error: no default value for X
Y de2 {}; // OK: use default constructor
X∗ p {nullptr};
X var {2}; // initialize local variable
p = new X{4}; // initialize object on free store
X a[] {1,2,3}; // initialize array elements
vector<X> v {1,2,3,4}; // initialize vector elements
}

Khởi tạo Đối tượng bằng Constructors ghi đè
• Dùng với khai báo constructor ghi đè (overload)
struct S {
S(const char∗);
S(double∗);
};
S s1 {"Napier"}; // S::S(const char*)
S s2 {new double{1.0}}; // S::S(double*);
S s3 {nullptr}; // ambiguous: S::S(const char*) or S::S(double*)?

Khởi tạo Đối tượng bằng Constructors có đối số
• Gọi tường minh constructor với đối số của nó
struct S1 {
int a,b; // no constructor
};
struct S2 {
int a,b;
S2(int a = 0, int b = 0) : a(aa), b(bb) {} // constructor
};
S1 x11(1,2); // error: no constructor
S1 x12 {1,2}; // OK: memberwise initialization
S1 x13(1); // error: no constructor
S1 x14 {1}; // OK: x14.b becomes 0
S2 x21(1,2); // OK: use constructor
S2 x22 {1,2}; // OK: use constructor
S2 x23(1); // OK: use constructor and one default argument
S2 x24 {1}; // OK: use constructor and one default argument

Khởi tạo Đối tượng bằng Constructors có đối số
Chú ý phân biệt khi 
• Khởi tạo đối tượng bằng danh sách các phần tử với
• Khởi tạo đối tượng bằng danh sách các đối số trong constructor
vector<int> v1 {77}; // one element with the value 77
vector<int> v2(77); // 77 elements with the default value 0

Ví dụ khác

vector<string> v4("Booh!"); // error: no constructor takes a string argument

vector<int∗> v6 {0,0}; // 2 int*s initialized to nullptr
vector<int∗> v7(0,0); // empty vector (v7.size()==0)
vector<int∗> v8; // empty vector (v7.size()==0)

• Là constructor không có đối số

• Cách dùng

• Cách dùng

Constructor mặc định cho kiểu Dữ liệu Tích hợp
• Các kiểu dữ liệu tích hợp có sẵn copy constructor và default constructor
• Giá trị khởi tạo của kiểu dữ liệu tích hợp
• 0 cho các kiểu số nguyên
• 0.0 cho các kiểu số dấu chấm động
• nullptr cho con trỏ
void f()
{
int a0; // uninitialized
int a1(); // function declaration (intended?)
int a {}; // a becomes 0
double d {}; // d becomes 0.0
char∗ p {}; // p becomes nullptr
int∗ p1 = new int; // uninitialized int
int∗ p2 = new int{}; // the int is initialized to 0
}

Constructor cho thành viên kiểu tham chiếu và const
• Các tham chiếu vào const phải được khởi tạo từ đầu

Khởi tạo Mảng, vector, các kiểu container
struct S1 { S1(); }; // has default constructor
struct S2 { S2(string); }; // no default constructor
S1 a1[10]; // OK: 10 default elements
S2 a2[10]; // error: cannot initialize elements
S2 a3[] { "alpha", "beta" }; // OK: two elements: S2{"alpha"}, S2{"beta"}

vector<S2> v2(10,""); // OK: 10 elements each initialized to S2{""}

Constructor với Danh sách các bộ Khởi tạo
• Đối số kiểu std::initializer<T> 
• Danh sách bộ khởi tạo phải đồng nhất
• Các phần tử của danh sách phải thuộc cùng kiểu, hoặc
• Có thể được chuyển kiểu sang T
• Ví dụ

f{1,2}; // error: function call () missing
years.insert({{"Bjarne","Stroustrup"},{1950, 1975, 1985}});

Luật Ưu tiên Chọn Constructor
• Chọn contructor có danh sách đối số tương hợp với giá trị tham số truyền vào lời gọi constructor
• Constructor mặc định và constructor với danh sách bộ khởi tạo được ưu tiên
• Nếu có cả constructor mặc định và constructor với danh sách bộ khởi tạo, ưu tiên constructor mặc định
• Nếu có constructor với danh sách bộ khởi tạo và constructor “thường”, ưu tiên chọn constructor với danh sách bộ khởi tạo

Luật Ưu tiên Chọn Constructor
• Ví dụ:
struct X {
X(initializer_list<int>);
X();
X(int);
};
X x0 {}; // empty list: default constructor or initializer-list constructor? (the default constructor)
X x1 {1}; // one integer: an int argument or a list of one element? (the initializer-list constructor)

Luật Ưu tiên Chọn Constructor
• Ví dụ: std::vector
vector<int> v1 {1}; // one element
vector<int> v2 {1,2}; // two elements
vector<int> v3 {1,2,3}; // three elements
vector<string> vs1 {"one"};
vector<string> vs2 {"one", "two"};
vector<string> vs3 {"one", "two", "three"};

Sử dụng initializer_list 
#include <initializer_list>
Hoặc
#include <vector>
#include <map>

Sử dụng initializer_list 
• Đối số kiểu initializer_list<T>
• là một chuỗi, được truy cập bằng các hàm thành viên begin(), end() và size()
• Được truyền vào hàm bằng giá trị
• Ví dụ: Duyệt phần tử của biến kiểu initializer_list<T> 
void f(initializer_list<int> args)
{
for (int i = 0; i!=args.size(); ++i)
cout << args.begin()[i] << "\n";
}

Sử dụng initializer_list 
Cách duyệt khác:
void f(initializer_list<int> args)
{
for (auto p=args.begin(); p!=args.end(); ++p)
cout << ∗p << "\n";
}
hoặc
void f(initializer_list<int> args)
{
for (auto x : args)
cout << x << "\n";
}

Sử dụng initializer_list 
• Các phần tử của đối số kiểu initializer_list<T> là bất biến.
int f(std::initializer_list<int> x, int val)
{
∗x.begin() = val; // error: attempt to change the value of an initializer-list element
return ∗x.begin(); // OK
}
void g()
{
for (int i=0; i!=10; ++i)
cout << f({1,2,3},i) << '\n';
}

Khởi tạo Trực tiếp và Khởi tạo bằng Sao chép
• Dùng toán tử khởi tạo {} 
• Áp dụng cho cả container và các phần tử của container
• Xét ví dụ với kiểu vector<vector<double>> 
vector<vector<double>> vs = {
{10,11,12,13,14}, // OK: vector of five elements
{10}, // OK: vector of one element
10, // error: vector<double>(int) is explicit
vector<double>{10,11,12,13}, // OK: vector of five elements
vector<double>{10}, // OK: vector of one element with value 10.0
vector<double>(10), // OK: vector of 10 elements with value 0.0
};

Khởi tạo Trực tiếp và Khởi tạo bằng Sao chép
• Một container có thể có cả constructor tường minh hoặc không tường minh
• Ví dụ:
• std::vector<int>(int) là tường minh
• std::vector<int>(initializer_list<int>) không tường minh
vector<double> v1(7); // OK: v1 has 7 elements; note: uses () rather than {}
vector<double> v2 = 9; // error: no conversion from int to vector
void f(const vector<double>&);
void g()
{
v1 = 9; // error: no conversion from int to vector
f(9); // error: no conversion from int to vector
}

Khởi tạo Trực tiếp và Khởi tạo bằng Sao chép
• Trong ví dụ trước, thay () bằng {} sẽ là cách sử dụng đúng
vector<double> v1 {7}; // OK: v1 has one element (with the value 7)
vector<double> v2 = {9}; // OK: v2 has one element (with the value 9)
void f(const vector<double>&);
void g()
{
v1 = {9}; // OK: v1 now has one element (with the value 9)
f({9}); // OK: f is called with the list {9}
}

};
• Đối tượng chỉ được tạo ra nếu constructor, và constructor ủy quyền hoàn thành
• Destructor sẽ không được gọi nếu đối tượng chưa được khởi tạo thành công

Sao chép và Di chuyển

Chuyển giá trị giữa các biến – Định nghĩa
• Sao chép (copy) : x = y 
• Kết quả: x và y có cùng giá trị của y trước khi thực hiện gán
• Chuyển (move): 
• Kết quả: x có giá trị (trước đó) của y
y mang trạng thái “đã được chuyển đi”. Sau đó, có thể thay đổi được

Chuyển giá trị giữa các biến – Nguyên tắc xử lý khi viết hàm
• Xử lý ngoại lệ
• Phép chuyển không thể gửi một ngoại lệ
• Phép sao chép có thể gửi ngoại lệ, do yêu cầu phải cấp tài nguyên cho biến
• Hiệu suất: phép chuyển (thường) hiểu quả hơn phép sao chép
• Viết hàm chuyển: nên giữ đối tượng nguồn ở trạng thái hợp lệ nhưng không xác định. 
• Hàm hủy chỉ dọn dẹp đối tượng ở trạng thái hợp lệ.

Sao chép
Phép sao chép của một lớp X được định nghĩa bởi hai thao tác:
• Khởi tạo chép - Copy constructor: X(const X&)
• Tạo ra bản sao của đối tượng và không làm thay đổi đối tượng này
• Chép qua phép gán – Copy assignment: X& operator=(const X&)

Sao chép – Ví dụ với lớp Matrix - Khai báo
template<class T>
class Matrix {
array<int,2> dim; // two dimensions
T∗ elem; // pointer to dim[0]*dim[1] elements of type T
public:
Matrix(int d1, int d2) :dim{d1,d2}, elem{new T[d1∗d2]} {} // simplified (no error handling)
int size() const { return dim[0]∗dim[1]; }
Matrix(const Matrix&); // copy constructor
Matrix& operator=(const Matrix&); // copy assignment
Matrix(Matrix&&); // move constructor
Matrix& operator=(Matrix&&); // move assignment
˜Matrix() { delete[] elem; }
// ...
};

Sao chép – Ví dụ với lớp Matrix - Cài đặt
template<class T>
Matrix:: Matrix(const Matrix& m) // copy constructor
: dim{m.dim},
elem{new T[m.size()]}
{
uninitialized_copy(m.elem,m.elem+m.size(),elem); // copy elements
}
template<class T>
Matrix& Matrix::operator=(const Matrix& m) // copy assignment
{
if (dim[0]!=m.dim[0] || dim[1]!=m.dim[1])
throw runtime_error("bad size in Matrix =");
copy(m.elem,m.elem+m.size(),elem); // copy elements
}

Sao chép các Thành phần Cơ sở
• Xem lớp cơ sở là một thành viên
• Để sao chép một đối tượng của lớp được kế thừa, nhớ phải sao chép cả các lớp cơ sở

// ...
};

Sao chép - Quy tắc thực hiện
Phép sao chép phải thỏa mãn:
1. Được khai báo kiểu đúng
2. Có ngữ nghĩa “sao chép” đúng
• Tính tương đương: Sau khi thực hiện phép gán x=y, 
• x và y có cùng giá trị
• Nếu tồn tại phép so sánh == thì x==y và f(x)==f(y) với f() bất kỳ phụ thuộc vào giá trị của x và y
• Tính độc lập: Sau khi thực hiện phép gán x=y, 
• Các phép toán trên x không nên ngầm thay đổi giá trị của y, i.e., f(x) không làm thay đổi giá trị của y miễn là f(x) không tham chiếu đến y
• Ví dụ: int và vector là hai lớp có phép sao chép như trên. 

Sao chép - Quy tắc thực hiện
Chú ý khi thực hiện sao chép đối tượng có chứa các con trỏ
• Phép sao chép mặc định thực hiện chép thành viên con trỏ, nhưng không chép đối tượng mà con trỏ trỏ đến. 
struct S {
int∗ p; // a pointer
};
S x {new int{0}};
void f()
{
S y {x}; // “copy” x
∗y.p = 1; // change y; affects x
∗x.p = 2; // change x; affects y
delete y.p; // affects x and y
y.p = new int{3}; // OK: change y; does not affect x
∗x.p = 4; // oops: write to deallocated memory
}

Sao chép – Sao chép “nông” vs. “sâu”
Với sao chép “nông” - shallow copy – hai đối tượng có cùng trạng thái được chia sẻ.
->Dễ gây lỗi.
Sao chép “sâu”: sao chép toàn bộ trạng thái của một đối tượng.
->nên thay thế bằng toán tử “di chuyển”
struct S {

};
S x {new int{0}};
void f()
{
S y {x}; // “copy” x
∗y.p = 1; // change y; affects x
∗x.p = 2; // change x; affects y

y.p = new int{3}; // OK: change y; does not affect x
∗x.p = 4; // oops: write to deallocated memory
}

Sao chép - Cắt lát (Slicing)
Con trỏ trỏ đến lớp dẫn xuất sẽ ngầm chuyển đổi thành con trỏ trỏ đến lớp cơ sở được khai báo public của nó. 
->Cắt lát
Giải pháp:

Phép chuyển của một lớp có thể thực hiện qua hai thao tác:
• Khởi tạo chuyển – Move constructor
• Chuyển bằng phép gán - Move assignment

Di chuyển – Khởi tạo Chuyển

:dim{a.dim}, elem{a.elem} // grab a’s representation
{

a.elem = nullptr;
}

return ∗this;
}

Di chuyển – swap() 
template<class T>
void swap(T& a, T& b) // "perfect swap" (almost)
{
T tmp = std::move(a);
a = std::move(b);
b = std::move(tmp);
}
template<class T>
void swap(T& a, T& b)
{
const T tmp = a; // put a copy of a into tmp
a = b; // put a copy of b into a
b = tmp; // put a copy of tmp into b
};
void f(string& s1, string& s2,
vector<string>& vs1, vector<string>& vs2,

{

}

Ví dụ: Phép + ma trận
Matrix operator+(const Matrix& a, const Matrix& b)
// res[i][j] = a[i][j]+b[i][j] for each i and j
{
if (a.dim[0]!=b.dim[0] || a.dim[1]!=b.dim[1])
throw std::runtime_error("unequal Matrix sizes in +");
Matrix res{a.dim[0],a.dim[1]};
constexpr auto n = a.size();
for (int i = 0; i!=n; ++i)
res.elem[i] = a.elem[i]+b.elem[i];
return res;
}

Phép toán giữa các đối tượng Lớp
• Lớp có sẵn các cài đặc mặc định (do trình biên dịch) sau:
• Khởi tạo mặc định: X()
• Khởi tạo sao chép: X(const X&)
• Phép gán bằng sao chép: X& operator=(const X&)
• Khởi tạo chuyển: X(X&&)
• Phép gán bằng chuyển: X& operator=(X&&)
• Hủy: ~X()
• Nếu một constructor được khai báo tường minh, constructor mặc định sẽ không được sinh ra
• Nếu phép sao chép, phép chuyển, hoặc destructor được khai báo tường minh, các thành phần tương ứng sẽ không được trình biên dịch sinh ra
